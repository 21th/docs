## Installing XState V5 beta

[XState](xstate.mdx) is a visual state management and orchestration library for JavaScript and TypeScript.

Install the XState V5 beta using your preferred package manager:

<Tabs>
<TabItem value="yarn" label="yarn">

```bash
yarn add xstate@beta
```

</TabItem>

<TabItem value="npm" label="npm">

```bash
npm install xstate@beta
```

</TabItem>
</Tabs>

## Using XState

The following feedback form example uses TypeScript. TypeScript isn’t necessary for XState, but it is advantageous!

```ts
import { assign, createMachine } from 'xstate';

// TypeScript types
const types = {
  context: {} as { feedback: string },
  events: {} as
    | {
        type: 'feedback.good';
      }
    | {
        type: 'feedback.bad';
      }
    | {
        type: 'feedback.update';
        value: string;
      }
    | { type: 'submit' }
    | {
        type: 'close';
      }
    | { type: 'back' }
    | { type: 'restart' },
};

export const feedbackMachine = createMachine({
  id: 'feedback',
  // initial state
  initial: 'prompt',
  types,
  context: {
    feedback: '',
  },
  states: {
    prompt: {
      on: {
        'feedback.good': 'thanks',
        'feedback.bad': 'form',
      },
    },
    form: {
      on: {
        'feedback.update': {
          // feedback.update does not transition
          // it updates the feedback value in the context
          actions: assign({
            feedback: ({event}) => event.value,
          }),
        },
        back: { target: 'prompt' },
        // guarded transition
        submit: {
          guard: ({ context }) => context.feedback.length > 0,
          target: 'thanks',
        },
      },
    },
    submitting: {
      invoke: {
          // fromPromise is an XState logic creator
          src: fromPromise(submitFeedback),
          onDone: 'thanks'
      } 
    },
    thanks: {},
    closed: {
      on: {
        restart: 'prompt',
      },
    },
  },
  on: {
    // targeting a child state
    close: '.closed',
  },
});
```
